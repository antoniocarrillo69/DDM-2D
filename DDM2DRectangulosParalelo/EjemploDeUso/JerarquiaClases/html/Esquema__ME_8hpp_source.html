91  void generaRepartoCarga(int n)
92  {
93  int i, j;
94 
95  // Ajusta el número de procesadores a usar
96  if (n >= (np-1)) npu = np - 1;
97  else npu = n;
98 
99  // Arreglo para soportar el reparto de carga por cada esclavo
100  ta = new int [npu];
101  for (i = 0; i < npu; i++) ta[i] = 0;
102 
103  // Reparto de carga
104  j = 0, i = 0;
105  while (i < n)
106  {
107  ta[j] ++;
108  i++;
109  j++;
110  if (j >= npu) j = 0;
111  }
112 
113  // Visualiza la carga
114  for (i = 0; i < npu; i++) printf("\nId = %d ==> TAREAS = %d ",i+1,ta[i]);
115  printf("\n\nNumero de procesadores esclavos a usar = %d\n\n",npu);
116  fflush(stdout);
117 
118  // Avisa a cada esclavo la cantidad de tareas a soportar
119  for (i = 1; i<= npu; i++) MPI::COMM_WORLD.Send(&ta[i-1], 1, MPI::INT, i, 1);
120  j=0;
121  for (i = (npu + 1); i < np; i++) MPI::COMM_WORLD.Send(&j, 1, MPI::INT, i, 1);
122  }
